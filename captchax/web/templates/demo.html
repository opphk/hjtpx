CaptchaX SDK 使用示例

1. 基础使用

点击按钮显示滑块验证码

2. Promise 方式

CaptchaX.verify({
    type: 'slider',
    container: '#captcha-container'
}).then(function(result) {
    console.log('验证成功:', result);
    // 提交表单等操作
}).catch(function(error) {
    console.log('验证失败:', error);
});

3. 回调方式

CaptchaX.onSuccess(function(result) {
    console.log('验证成功:', result.token);
    document.getElementById('token').value = result.token;
});

CaptchaX.onError(function(error) {
    console.log('验证失败:', error.error);
});

CaptchaX.verify({ type: 'click' });

4. 嵌入指定容器

点击按钮后验证码将显示在这里


5. 自定义配置

CaptchaX.init({
    serverUrl: 'https://your-captchax-server.com',
    appId: 'your-app-id',
    timeout: 10000,
    retryAttempts: 3,
    loadingText: '加载中...',
    successText: '验证成功',
    failText: '验证失败'
});

CaptchaX.verify({
    type: 'slider',
    theme: 'dark',
    autoClose: true
});
