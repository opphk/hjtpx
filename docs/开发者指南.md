```go
    points := make([]TrajectoryPoint, 0, 10)
```

#### 无感验证集成

```javascript
class SeamlessCaptcha {
    constructor() {
        this.trustLevel = null;
        this.fingerprint = null;
    }
    
    async init() {
        // 收集设备指纹
        this.fingerprint = await this.collectFingerprint();
        
        // 检查信任级别
        const result = await this.checkTrustLevel();
        
        if (result.requires_captcha) {
            // 需要显示验证码
            this.showCaptcha();
        } else {
            // 直接通过
            this.onTrustVerified(result);
        }
    }
    
    async collectFingerprint() {
        const canvas = document.createElement('canvas');
        const ctx = canvas.getContext('2d');
        ctx.textBaseline = 'top';
        ctx.font = '14px Arial';
        ctx.fillText('fingerprint', 2, 2);
        
        return canvas.toDataURL().hash();
    }
    
    async checkTrustLevel() {
        const response = await fetch('http://localhost:8080/api/v1/seamless/check', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify({
                device_fingerprint: this.fingerprint
            })
        });
        
        return await response.json();
    }
    
    onTrustVerified(result) {
        console.log('信任验证通过，等级:', result.trust_level);
        console.log('风险分数:', result.risk_score);
    }
}
```

```go
        return
    }
    
    // 2. 执行登录逻辑
    // ...
    
    // 3. 返回结果
    json.NewEncoder(w).Encode(map[string]interface{}{
        "code": 0,
        "message": "登录成功",
    })
}

func verifyCaptchaToken(token string) (bool, error) {
    url := "http://localhost:8080/api/v1/captcha/verify-token"
    resp, err := http.Post(url, "application/json", 
        strings.NewReader(fmt.Sprintf(`{"token":"%s"}`, token)))
    
    if err != nil {
        return false, err
    }
    defer resp.Body.Close()
    
    var result map[string]interface{}
    json.NewDecoder(resp.Body).Decode(&result)
    
    return result["success"] == true, nil
}
```

```go
func CaptchaMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        // 跳过不需要验证的路径
        skipPaths := []string{"/health", "/static/"}
        for _, path := range skipPaths {
            if strings.HasPrefix(r.URL.Path, path) {
                next.ServeHTTP(w, r)
                return
            }
        }
        
        // 从请求中获取验证码Token
        token := r.Header.Get("X-Captcha-Token")
        if token == "" {

            http.Error(w, "验证码验证失败", http.StatusUnauthorized)
            return
        }
        
        next.ServeHTTP(w, r)
    })
}
```

```go
func signRequest(req *http.Request, secret string) {
    timestamp := time.Now().Unix()
    message := fmt.Sprintf("%s%d", req.URL.Path, timestamp)
    
    h := hmac.New(sha256.New, []byte(secret))
    h.Write([]byte(message))
    signature := hex.EncodeToString(h.Sum(nil))
    
    req.Header.Set("X-Timestamp", strconv.FormatInt(timestamp, 10))
    req.Header.Set("X-Signature", signature)
}

// 验证签名
func verifySignature(secret, signature, path string, timestamp int64) bool {
    if time.Now().Unix()-timestamp > 300 {
        return false // 签名过期
    }
    
    message := fmt.Sprintf("%s%d", path, timestamp)
    h := hmac.New(sha256.New, []byte(secret))
    h.Write([]byte(message))
    expected := hex.EncodeToString(h.Sum(nil))
    
    return hmac.Equal([]byte(signature), []byte(expected))
}
```

```go
// 本地限流器
type RateLimiter struct {
    mu       sync.Mutex
    requests map[string][]time.Time
    limit    int
    window   time.Duration
}

func (r *RateLimiter) Allow(key string) bool {
    r.mu.Lock()
    defer r.mu.Unlock()
    
    now := time.Now()
    cutoff := now.Add(-r.window)
    
    var valid []time.Time
    for _, t := range r.requests[key] {
        if t.After(cutoff) {
            valid = append(valid, t)
        }
    }
    
    if len(valid) >= r.limit {
        r.requests[key] = valid
        return false
    }
    
    r.requests[key] = append(valid, now)
    return true
}
```

```go
    var err error

        err = fn()
```

```go
type CircuitBreaker struct {
    failures    int
    threshold   int
    timeout     time.Duration
    lastFailure time.Time
    state       string // closed, open, half-open
}

func (cb *CircuitBreaker) Call(fn func() error) error {
    if cb.state == "open" {
        if time.Since(cb.lastFailure) > cb.timeout {
            cb.state = "half-open"
        } else {
            return fmt.Errorf("熔断器开启，拒绝请求")
        }
    }
    
    err := fn()
    if err != nil {
        cb.failures++
        cb.lastFailure = time.Now()
        
        if cb.failures >= cb.threshold {
            cb.state = "open"
        }
        
        return err
    }
    
    cb.failures = 0
    cb.state = "closed"
    return nil
}
```
