## 常见问题解答（FAQ）

### 1. 验证码相关问题

#### Q: 验证码Session过期怎么办？
**A:** 验证码Session默认5分钟过期。过期后需要重新生成验证码。建议在前端实现自动刷新机制：

```javascript
// 监听验证码过期事件
captchaWidget.on('expired', () => {
    showToast('验证码已过期，正在刷新...');
    refreshCaptcha();
});

// 设置自动刷新
setInterval(() => {
    if (captchaWidget.isVisible() && !captchaWidget.isCompleted()) {
        const elapsed = Date.now() - captchaWidget.getStartTime();
        if (elapsed > 4 * 60 * 1000) { // 4分钟后开始检查
            captchaWidget.refresh();
        }
    }
}, 60000);
```

#### Q: 如何提高验证码通过率？
**A:** 
1. 确保轨迹数据真实且自然
2. 允许一定的容差范围
3. 提供清晰的用户引导
4. 根据设备类型选择合适的验证码类型

```go
// 根据设备类型选择验证码
func selectCaptchaType(device string) string {
    switch {
    case device == "mobile":
        return "emoji" // 移动端适合Emoji
    case device == "tablet":
        return "gesture" // 平板适合手势
    default:
        return "slider" // 桌面端使用滑块
    }
}
```

#### Q: 验证码验证失败的原因有哪些？
**A:** 
- 滑动位置偏差过大
- 轨迹数据被判定为机器人行为
- 环境检测发现异常（使用自动化工具）
- Session已过期
- 验证次数超限

func signRequest(req *http.Request, secret string) {
    timestamp := time.Now().Unix()
    message := fmt.Sprintf("%s%d%s", req.URL.Path, timestamp, req.Method)
    
    mac := hmac.New(sha256.New, []byte(secret))
    mac.Write([]byte(message))
    signature := hex.EncodeToString(mac.Sum(nil))
    
    req.Header.Set("X-Timestamp", strconv.FormatInt(timestamp, 10))
    req.Header.Set("X-Signature", signature)
}
```

func handleLogin(w http.ResponseWriter, r *http.Request) {
    var req LoginRequest
    json.NewDecoder(r.Body).Decode(&req)
    
    // 1. 服务端验证token
    verifyURL := fmt.Sprintf("%s/api/v1/captcha/verify-token", config.BaseURL)
    resp, err := http.Post(verifyURL, "application/json",
        strings.NewReader(fmt.Sprintf(`{"token":"%s"}`, req.CaptchaToken)))
    
    if err != nil || resp.StatusCode != http.StatusOK {
        http.Error(w, "验证码验证失败", http.StatusBadRequest)
        return
    }
    
    var result map[string]interface{}
    json.NewDecoder(resp.Body).Decode(&result)
    
    if result["success"] != true {
        http.Error(w, "验证码验证失败", http.StatusBadRequest)
        return
    }
    
    // 2. 检查风险分数
    riskScore := result["risk_score"].(float64)
    if riskScore > 50 {
        log.Printf("高风险登录: %.2f", riskScore)
        // 可以要求额外验证
    }
    
    // 3. 执行登录
    // ...
}
```

func signRequest(req *http.Request, secret string) {
    timestamp := time.Now().Unix()
    message := fmt.Sprintf("%s%d", req.URL.Path, timestamp)
    
    h := hmac.New(sha256.New, []byte(secret))
    h.Write([]byte(message))
    signature := hex.EncodeToString(h.Sum(nil))
    
    req.Header.Set("X-Timestamp", strconv.FormatInt(timestamp, 10))
    req.Header.Set("X-Signature", signature)
}

// 验证签名
func verifySignature(secret, signature, path string, timestamp int64) bool {
    if time.Now().Unix()-timestamp > 300 {
        return false // 签名过期
    }
    
    message := fmt.Sprintf("%s%d", path, timestamp)
    h := hmac.New(sha256.New, []byte(secret))
    h.Write([]byte(message))
    expected := hex.EncodeToString(h.Sum(nil))
    
    return hmac.Equal([]byte(signature), []byte(expected))
}

// 本地限流器
type RateLimiter struct {
    mu       sync.Mutex
    requests map[string][]time.Time
    limit    int
    window   time.Duration
}

func (r *RateLimiter) Allow(key string) bool {
    r.mu.Lock()
    defer r.mu.Unlock()
    
    now := time.Now()
    cutoff := now.Add(-r.window)
    
    var valid []time.Time
    for _, t := range r.requests[key] {
        if t.After(cutoff) {
            valid = append(valid, t)
        }
    }
    
    if len(valid) >= r.limit {
        r.requests[key] = valid
        return false
    }
    
    r.requests[key] = append(valid, now)
    return true
}

    var err error
    for i := 0; i < maxRetries; i++ {
        err = fn()
        if err == nil {

type CircuitBreaker struct {
    failures    int
    threshold   int
    timeout     time.Duration
    lastFailure time.Time
    state       string // closed, open, half-open
}

func (cb *CircuitBreaker) Call(fn func() error) error {
    if cb.state == "open" {
        if time.Since(cb.lastFailure) > cb.timeout {
            cb.state = "half-open"
        } else {
            return fmt.Errorf("熔断器开启，拒绝请求")
        }
    }
    
    err := fn()
    if err != nil {
        cb.failures++
        cb.lastFailure = time.Now()
        
        if cb.failures >= cb.threshold {
            cb.state = "open"
        }
        
        return err
    }
    
    cb.failures = 0
    cb.state = "closed"
    return nil
}
