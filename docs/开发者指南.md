## SDK使用指南

### Go SDK

#### 安装

```bash
go get github.com/hjtpx/hjtpx/sdk/go
```

#### 基础配置

```go
package main

import (
    "github.com/hjtpx/hjtpx/sdk/go"
)

func main() {
    // 方式一：简单客户端（适合开发测试）
    client := captcha.NewClient(
        captcha.WithAPIKey("your-api-key"),
        captcha.WithAPISecret("your-api-secret"),
        captcha.WithEndpoint("http://localhost:8080"),
    )
    defer client.Close()

    // 方式二：高级客户端（推荐生产环境）
    cfg := &captcha.Config{
        BaseURL:        "http://localhost:8080",
        MaxRetries:     3,
        HTTPTimeout:    10 * time.Second,
        MaxIdleConns:   10,
        MaxOpenConns:   100,
        RetryDelay:     100 * time.Millisecond,
    }
    
    client := captcha.NewCaptchaClient("app-id", "app-secret", cfg)
    defer client.Close()
}
```

#### 轨迹数据收集

```go
type TrajectoryPoint struct {
    X        float64
    Y        float64
    Timestamp int64
}

func generateTrajectory(targetY int) []TrajectoryPoint {
    baseTime := time.Now().UnixMilli()
    points := make([]TrajectoryPoint, 0, 10)
    
    for i := 0; i < 10; i++ {
        points = append(points, TrajectoryPoint{
            X:        float64(i * 16),
            Y:        float64(targetY) + float64(i%3-1) * 2,
            Timestamp: baseTime + int64(i*50),
        })
    }
    
    return points
}
```

```typescript
import { CaptchaClient, CaptchaConfig, RetryConfig } from '@hjtpx/captcha-sdk';

const config: CaptchaConfig = {
    baseUrl: 'http://localhost:8080',
    apiKey: 'your-api-key',
    apiSecret: 'your-api-secret',
    timeout: 30000,
    maxConnections: 100,

        }
    }

const client = new CaptchaClient(config);
```

```typescript
// 类型定义
interface CaptchaConfig {
    baseUrl: string;
    apiKey: string;
    apiSecret: string;
    timeout?: number;
    maxConnections?: number;
    retryConfig?: RetryConfig;
}

interface RetryConfig {
    maxRetries: number;
    initialDelayMs: number;
    maxDelayMs: number;
    shouldRetry?: (error: Error) => boolean;
}

interface TrajectoryPoint {
    x: number;
    y: number;
    timestamp: number;
}
```

```go
    limit    int
    window   time.Duration

func (r *RateLimiter) Allow(key string) bool {
    r.mu.Lock()
    defer r.mu.Unlock()
    
    now := time.Now()
    cutoff := now.Add(-r.window)
    
    var valid []time.Time

    }

        return false
    }

    return true
```
```go
}

func (cb *CircuitBreaker) Call(fn func() error) error {
    if cb.state == "open" {
        if time.Since(cb.lastFailure) > cb.timeout {
            cb.state = "half-open"
        } else {
            return fmt.Errorf("熔断器开启，拒绝请求")
        }
    }
    
    err := fn()
    if err != nil {
        cb.failures++
        cb.lastFailure = time.Now()
        
        if cb.failures >= cb.threshold {
            cb.state = "open"
        }
        
        return err
    }
    
    cb.failures = 0
    cb.state = "closed"
    return nil
}
```

```go
type HealthChecker struct {

    redis *redis.Client

func (h *HealthChecker) Check() error {
    // 检查数据库
    if err := h.db.Ping(); err != nil {
        return fmt.Errorf("数据库连接失败: %w", err)
    }
    
    // 检查Redis
    if err := h.redis.Ping().Err(); err != nil {
        return fmt.Errorf("Redis连接失败: %w", err)
    }
    
    return nil
}

func (h *HealthChecker) ServeHTTP(w http.ResponseWriter, r *http.Request) {
    if err := h.Check(); err != nil {
        w.WriteHeader(http.StatusServiceUnavailable)
        json.NewEncoder(w).Encode(map[string]interface{}{
            "status": "unhealthy",
            "error":  err.Error(),
        })
        return
    }
    
    w.WriteHeader(http.StatusOK)
    json.NewEncoder(w).Encode(map[string]interface{}{
        "status": "healthy",
    })
}
```
