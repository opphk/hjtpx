🔢 验证码API

滑块验证码

GET /api/v1/captcha/slider

获取滑块验证码

参数类型必填说明
widthint否图片宽度，默认320
heightint否图片高度，默认160
toleranceint否容差值，默认8
 
{

    "message": "success",
    "data": {

        "image_url": "data:image/png;base64,...",
        "puzzle_url": "data:image/png;base64,...",
        "hint_url": "data:image/png;base64,...",
        "shape": 1,
        "secret_y": 100,
        "image_width": 320,
        "image_height": 160
    }
}
POST /api/v1/captcha/verify

验证滑块验证码

参数类型必填说明
session_idstring是验证码会话ID
xint是X坐标
yint否Y坐标
trajectoryarray否滑动轨迹
 
{
    "session_id": "slider_abc123def456",
    "x": 185,
    "y": 100,
    "trajectory": [
        {"x": 0, "y": 100, "t": 0},
        {"x": 185, "y": 100, "t": 800}
    ]
}

📦 SDK示例

Go SDK 使用示例

package main

import (
    "fmt"
    "time"
    "your-project/sdk/go/captcha"
)

func main() {
    // 创建客户端
    client := captcha.NewClient("http://localhost:8080")
    
    // 获取滑块验证码
    captcha, err := client.GetSliderCaptcha(320, 160, 8)
    if err != nil {
        panic(err)
    }
    fmt.Printf("Session ID: %s\n", captcha.SessionID)
    
    // 验证验证码
    result, err := client.VerifyCaptcha(&captcha.VerifyCaptchaRequest{
        SessionID: captcha.SessionID,
        X:         185,
        Y:         captcha.SecretY,
    })
    fmt.Printf("Success: %v\n", result.Success)
}

JavaScript SDK 使用示例

// 创建客户端
const client = new CaptchaClient('http://localhost:8080');

// 获取滑块验证码
const captcha = await client.getSliderCaptcha({
    width: 320,
    height: 160
});

// 记录轨迹
const recorder = client.recordTrajectory();

// 验证
const result = await client.verifyCaptcha({
    session_id: captcha.session_id,
    x: 185,
    y: captcha.secret_y,
    trajectory: recorder.stop()
});

console.log('验证结果:', result);

Python SDK 使用示例

from captcha import CaptchaClient, TrajectoryPoint

# 创建客户端
client = CaptchaClient('http://localhost:8080')

# 获取滑块验证码
captcha = client.get_slider_captcha(width=320, height=160, tolerance=8)
print(f"Session ID: {captcha.session_id}")

# 生成轨迹
trajectory = [
    TrajectoryPoint(x=0, y=captcha.secret_y, t=0),
    TrajectoryPoint(x=185, y=captcha.secret_y, t=800),
]

# 验证
result = client.verify_captcha(
    session_id=captcha.session_id,
    x=185,
    y=captcha.secret_y,
    trajectory=trajectory
)

print(f"Success: {result.success}")
