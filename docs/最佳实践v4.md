## 1. 基础集成案例

### 案例 1.1: Web 端快速集成

**场景描述**

某电商平台需要在登录、注册、评论等场景集成验证码，要求快速上线且用户体验良好。

**解决方案**

```html
<!DOCTYPE html>
<html>
<head>
    <title>登录页面</title>
    <script src="https://cdn.hjtpx.com/captcha.min.js"></script>
</head>
<body>
    <form id="loginForm">
        <input type="text" id="username" placeholder="用户名">
        <input type="password" id="password" placeholder="密码">

        <!-- 验证码容器 -->
        <div id="captchaContainer"></div>

        <button type="submit" id="submitBtn">登录</button>
    </form>

    <script>
        // 初始化 HJTPX SDK
        const captcha = new HJTPX.Captcha({
            appId: 'your_app_id',
            mode: 'inline',  // 嵌入式模式
            container: '#captchaContainer',
            theme: 'light',
            lang: 'zh-CN'
        });

        // 渲染验证码
        captcha.render();

        // 监听验证成功事件
        captcha.on('verifySuccess', (result) => {
            console.log('验证成功:', result);
            // 提交表单
            document.getElementById('submitBtn').disabled = false;
        });

        // 监听验证失败事件
        captcha.on('verifyFail', (error) => {
            console.error('验证失败:', error);
            document.getElementById('submitBtn').disabled = true;
        });

        // 表单提交
        document.getElementById('loginForm').addEventListener('submit', async (e) => {
            e.preventDefault();

            const captchaToken = captcha.getToken();
            if (!captchaToken) {
                alert('请先完成验证码');
                return;
            }

            const response = await fetch('/api/login', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify({
                    username: document.getElementById('username').value,
                    password: document.getElementById('password').value,
                    captchaToken: captchaToken
                })
            });

            const data = await response.json();
            if (data.success) {
                window.location.href = '/dashboard';
            }
        });
    </script>
</body>
</html>
```

**效果评估**

- 上线时间：1天
- 验证通过率：96.5%
- 用户体验评分：4.8/5

### 案例 2.1: 高并发场景优化

**场景描述**

某票务系统在演出开票时面临巨大流量冲击，需要验证码服务在高并发下保持稳定。

**问题分析**

1. 验证码生成成为瓶颈
2. 验证请求堆积
3. 系统响应延迟增加

**解决方案**

```go
// 高性能验证码服务配置
type HighPerformanceConfig struct {
    // 连接池优化
    MaxOpenConns    int = 100
    MaxIdleConns    int = 50
    ConnMaxLifetime time.Duration = 30 * time.Minute

    // Redis 缓存优化
    RedisPoolSize   int = 200
    RedisMinIdle    int = 20

    // 并发控制
    WorkerPoolSize  int = 100
    QueueSize       int = 10000

    // 缓存策略
    CacheEnabled    bool = true
    CacheTTL        time.Duration = 5 * time.Minute
}

// 验证码服务实现
type HighPerformanceCaptchaService struct {
    cache      *redis.Client
    db         *sql.DB
    workerPool *WorkerPool
}

func NewHighPerformanceCaptchaService(cfg *HighPerformanceConfig) (*HighPerformanceCaptchaService, error) {
    // 初始化 Redis 连接池
    redisClient := redis.NewClient(&redis.Options{
        Addr:         cfg.RedisAddr,
        PoolSize:     cfg.RedisPoolSize,
        MinIdleConns: cfg.RedisMinIdle,
        ReadTimeout:  3 * time.Second,
        WriteTimeout: 3 * time.Second,
    })

    // 初始化数据库连接池
    db.SetMaxOpenConns(cfg.MaxOpenConns)
    db.SetMaxIdleConns(cfg.MaxIdleConns)
    db.SetConnMaxLifetime(cfg.ConnMaxLifetime)

    // 创建工作池
    workerPool := NewWorkerPool(cfg.WorkerPoolSize, cfg.QueueSize)

    return &HighPerformanceCaptchaService{
        cache:      redisClient,
        db:         db,
        workerPool: workerPool,
    }, nil
}

// 生成验证码 - 使用异步处理
func (s *HighPerformanceCaptchaService) GenerateAsync(ctx context.Context, req *GenerateRequest) (<-chan *GenerateResponse, error) {
    responseCh := make(chan *GenerateResponse, 1)

    s.workerPool.Submit(Task{
        Context: ctx,
        Handler: func(ctx context.Context) error {
            // 检查缓存
            cacheKey := fmt.Sprintf("captcha:%s:%s", req.AppID, req.Type)
            cached, err := s.cache.Get(ctx, cacheKey).Result()
            if err == nil && cached != "" {
                var resp GenerateResponse
                if json.Unmarshal([]byte(cached), &resp) == nil {
                    responseCh <- &resp
                    return nil
                }
            }

            // 生成新验证码
            resp, err := s.generateCaptcha(ctx, req)
            if err != nil {
                return err
            }

            // 更新缓存
            respJSON, _ := json.Marshal(resp)
            s.cache.Set(ctx, cacheKey, respJSON, 5*time.Minute)

            responseCh <- resp
            return nil
        },
    })

    return responseCh, nil
}

// 批量生成验证码
func (s *HighPerformanceCaptchaService) GenerateBatch(ctx context.Context, req *BatchGenerateRequest) (*BatchGenerateResponse, error) {
    results := make([]*GenerateResponse, len(req.Requests))

    // 使用 goroutine 并发生成
    var wg sync.WaitGroup
    for i, r := range req.Requests {
        wg.Add(1)
        go func(index int, request *GenerateRequest) {
            defer wg.Done()
            resp, _ := s.Generate(ctx, request)
            results[index] = resp
        }(i, r)
    }

    wg.Wait()

    return &BatchGenerateResponse{
        Results: results,
        Total:   len(results),
    }, nil
}
```

```javascript
class LazyCaptchaLoader {
    constructor() {
        this.loaded = false;
        this.loading = false;
    }

    async load() {
        if (this.loaded) return;
        if (this.loading) return;

        this.loading = true;

        // 动态导入
        await Promise.all([
            import('https://cdn.hjtpx.com/captcha-core.min.js'),
            import('https://cdn.hjtpx.com/captcha-styles.min.css')
        ]);

        // 预连接关键资源
        const preconnectUrls = [
            'https://cdn.hjtpx.com',
            'https://api.hjtpx.com'
        ];

        preconnectUrls.forEach(url => {
            const link = document.createElement('link');
            link.rel = 'preconnect';
            link.href = url;
            document.head.appendChild(link);
        });

        this.loaded = true;
        this.loading = false;
    }

    // Web Worker 初始化
    async initWorker() {
        const workerCode = `
            self.onmessage = async (e) => {
                if (e.data.type === 'init') {
                    // Worker 初始化
                    self.postMessage({ type: 'ready' });
                }
            };
        `;

        const blob = new Blob([workerCode], { type: 'application/javascript' });
        this.worker = new Worker(URL.createObjectURL(blob));
    }
}
```

## 3. 安全加固案例

### 案例 3.1: 防止暴力破解

**场景描述**

某金融系统需要防止恶意用户通过暴力破解方式尝试登录。

**解决方案**

```go
type BruteForceProtection struct {
    redis       *redis.Client
    maxAttempts int
    lockoutTime time.Duration
    threshold   int
}

func NewBruteForceProtection(redis *redis.Client) *BruteForceProtection {
    return &BruteForceProtection{
        redis:       redis,
        maxAttempts: 5,
        lockoutTime: 15 * time.Minute,
        threshold:   3,
    }
}

// 检查是否被锁定
func (b *BruteForceProtection) IsLocked(ip string) (bool, time.Duration) {
    key := fmt.Sprintf("bruteforce:lockout:%s", ip)
    ttl, err := b.redis.TTL(key).Result()
    if err != nil || ttl < 0 {
        return false, 0
    }
    return true, ttl
}

// 记录失败尝试
func (b *BruteForceProtection) RecordFailedAttempt(ip string) error {
    key := fmt.Sprintf("bruteforce:attempts:%s", ip)

    pipe := b.redis.Pipeline()
    pipe.Incr(key)
    pipe.Expire(key, b.lockoutTime)
    _, err := pipe.Exec()

    if err != nil {
        return err
    }

    // 检查是否达到锁定阈值
    attempts, _ := b.redis.Get(key).Int()
    if attempts >= b.maxAttempts {
        lockoutKey := fmt.Sprintf("bruteforce:lockout:%s", ip)
        b.redis.Set(lockoutKey, "1", b.lockoutTime)
    }

    return nil
}

// 清除失败记录
func (b *BruteForceProtection) ClearAttempts(ip string) error {
    key := fmt.Sprintf("bruteforce:attempts:%s", ip)
    return b.redis.Del(key).Err()
}

// API 中间件
func BruteForceMiddleware(protection *BruteForceProtection) gin.HandlerFunc {
    return func(c *gin.Context) {
        ip := c.ClientIP()

        if locked, ttl := protection.IsLocked(ip); locked {
            c.JSON(http.StatusTooManyRequests, gin.H{
                "success": false,
                "error": gin.H{
                    "code":    "ACCOUNT_LOCKED",
                    "message": fmt.Sprintf("账户已锁定，请在 %v 后重试", ttl),
                },
            })
            c.Abort()
            return
        }

        c.Next()

        // 如果登录失败
        if c.Writer.Status() == http.StatusUnauthorized {
            protection.RecordFailedAttempt(ip)
        }

        // 如果登录成功
        if c.Writer.Status() == http.StatusOK {
            protection.ClearAttempts(ip)
        }
    }
}
```

### 案例 3.2: API 密钥安全存储

**场景描述**

需要在服务端安全存储和管理 API 密钥。

**解决方案**

```go
type SecureKeyStore struct {
    kms       *KMSClient
    db        *sql.DB
    encryptor *Encryptor
}

func NewSecureKeyStore(kms *KMSClient, db *sql.DB) *SecureKeyStore {
    return &SecureKeyStore{
        kms:   kms,
        db:    db,
        encryptor: NewEncryptor(kms),
    }
}

// 生成新密钥
func (s *SecureKeyStore) GenerateKey(appID string) (*APIKey, error) {
    // 生成随机密钥
    keyBytes := make([]byte, 32)
    if _, err := rand.Read(keyBytes); err != nil {
        return nil, err
    }
    key := base64.StdEncoding.EncodeToString(keyBytes)

    // 生成密钥哈希
    hash := sha256.Sum256([]byte(key))

    // 加密存储
    encryptedKey, err := s.encryptor.Encrypt(keyBytes)
    if err != nil {
        return nil, err
    }

    // 生成密钥 ID
    keyID := uuid.New().String()

    // 存储到数据库
    apiKey := &APIKey{
        ID:           keyID,
        AppID:        appID,
        EncryptedKey: encryptedKey,
        KeyHash:      hex.EncodeToString(hash[:]),
        CreatedAt:    time.Now(),
        Status:       Active,
    }

    if err := s.db.Create(apiKey).Error; err != nil {
        return nil, err
    }

    // 返回明文密钥（仅此时可见）
    return &APIKey{
        ID:        keyID,
        Key:       key, // 一次性返回
        AppID:     appID,
        CreatedAt: apiKey.CreatedAt,
    }, nil
}

// 验证密钥
func (s *SecureKeyStore) ValidateKey(keyID, key string) (bool, error) {
    var apiKey APIKey
    if err := s.db.Where("id = ? AND status = ?", keyID, Active).First(&apiKey).Error; err != nil {
        return false, err
    }

    // 计算哈希
    hash := sha256.Sum256([]byte(key))

    // 比对
    return hex.EncodeToString(hash[:]) == apiKey.KeyHash, nil
}

// 获取解密后的密钥（仅内部使用）
func (s *SecureKeyStore) GetDecryptedKey(keyID string) ([]byte, error) {
    var apiKey APIKey
    if err := s.db.Where("id = ?", keyID).First(&apiKey).Error; err != nil {
        return nil, err
    }

    return s.encryptor.Decrypt(apiKey.EncryptedKey)
}

// 轮换密钥
func (s *SecureKeyStore) RotateKey(keyID string) (*APIKey, error) {
    var apiKey APIKey
    if err := s.db.Where("id = ?", keyID).First(&apiKey).Error; err != nil {
        return nil, err
    }

    // 标记旧密钥为过期
    apiKey.Status = Rotating
    apiKey.ExpiresAt = time.Now().Add(24 * time.Hour)
    s.db.Save(apiKey)

    // 生成新密钥
    return s.GenerateKey(apiKey.AppID)
}
```

### 案例 4.1: 智能组合验证码

**场景描述**

某在线考试系统需要根据用户行为动态选择验证码类型，平衡安全性和用户体验。

**解决方案**

```javascript
class AdaptiveCaptchaSelector {
    constructor(config) {
        this.riskEngine = new RiskEngine(config.riskThreshold);
        this.captchaInstances = new Map();
    }

    async selectCaptcha(userContext) {
        // 1. 风险评估
        const riskScore = await this.riskEngine.assess(userContext);

        // 2. 根据风险级别选择验证码
        const captchaType = this.determineCaptchaType(riskScore);

        // 3. 获取验证码实例
        let captcha = this.captchaInstances.get(captchaType);
        if (!captcha) {
            captcha = await this.createCaptchaInstance(captchaType);
            this.captchaInstances.set(captchaType, captcha);
        }

        // 4. 根据风险级别调整难度
        captcha.setDifficulty(this.getDifficultyForRisk(riskScore));

        return {
            type: captchaType,
            instance: captcha,
            riskScore: riskScore
        };
    }

    determineCaptchaType(riskScore) {
        if (riskScore < 20) {
            // 低风险：使用简单的图片验证码
            return 'image';
        } else if (riskScore < 50) {
            // 中低风险：使用滑动验证码
            return 'slider';
        } else if (riskScore < 75) {
            // 中高风险：使用点击验证码
            return 'click';
        } else {
            // 高风险：使用组合验证码
            return 'combo';
        }
    }

    getDifficultyForRisk(riskScore) {
        if (riskScore < 30) {
            return { tolerance: 15, timeLimit: 60 };
        } else if (riskScore < 60) {
            return { tolerance: 10, timeLimit: 45 };
        } else {
            return { tolerance: 5, timeLimit: 30 };
        }
    }
}

class RiskEngine {
    constructor(threshold) {
        this.threshold = threshold;
        this.history = new Map();
    }

    async assess(context) {
        let score = 0;

        // 1. 地理位置分析
        score += await this.analyzeGeoLocation(context);

        // 2. 设备指纹分析
        score += this.analyzeFingerprint(context.fingerprint);

        // 3. 历史行为分析
        score += await this.analyzeHistory(context.userId);

        // 4. 网络环境分析
        score += this.analyzeNetwork(context.network);

        // 5. 实时行为分析
        score += await this.analyzeBehavior(context.behavior);

        return Math.min(100, score);
    }

    async analyzeFingerprint(fingerprint) {
        const riskFactors = {
            newDevice: fingerprint.isNew ? 20 : 0,
            suspiciousPlugins: fingerprint.pluginCount > 50 ? 15 : 0,
            noWebGL: !fingerprint.hasWebGL ? 10 : 0,
            automationDetected: fingerprint.isAutomated ? 30 : 0,
        };

        return Object.values(riskFactors).reduce((a, b) => a + b, 0);
    }

    async analyzeHistory(userId) {
        const stats = await this.getUserStats(userId);
        let score = 0;

        // 频繁失败
        if (stats.failedAttempts > 10) score += 20;

        // 异常登录时间
        if (this.isUnusualHour(stats.lastLogin)) score += 10;

        // IP 变化频繁
        if (stats.uniqueIPs > 10) score += 15;

        return score;
    }
}
```

### 案例 4.2: 行为生物特征认证

**场景描述**

某银行系统希望使用鼠标轨迹、键盘敲击等生物特征进行持续身份验证。

**解决方案**

```go
type BehaviorBiometricsService struct {
    trajectoryAnalyzer *TrajectoryAnalyzer
    keyboardAnalyzer  *KeyboardAnalyzer
    ensemble          *BiometricEnsemble
    db                *sql.DB
}

func NewBehaviorBiometricsService(cfg *Config) *BehaviorBiometricsService {
    return &BehaviorBiometricsService{
        trajectoryAnalyzer: NewTrajectoryAnalyzer(cfg.modelPath),
        keyboardAnalyzer:   NewKeyboardAnalyzer(cfg.keyboardModel),
        ensemble:           NewBiometricEnsemble(),
        db:                cfg.db,
    }
}

// 采集鼠标轨迹
func (s *BehaviorBiometricsService) CollectTrajectory(userID string, events []TrajectoryEvent) error {
    features := s.trajectoryAnalyzer.ExtractFeatures(events)

    // 存储特征
    featureData := &TrajectoryFeature{
        UserID:         userID,
        Velocity:       features.Velocity,
        Acceleration:   features.Acceleration,
        Curvature:      features.Curvature,
        Direction:      features.Direction,
        PausePatterns:  features.PausePatterns,
        CollectedAt:    time.Now(),
    }

    return s.db.Create(featureData).Error
}

// 采集键盘行为
func (s *BehaviorBiometricsService) CollectKeyboard(userID string, events []KeyboardEvent) error {
    features := s.keyboardAnalyzer.ExtractFeatures(events)

    featureData := &KeyboardFeature{
        UserID:             userID,
        TypingSpeed:        features.TypingSpeed,
        FlightTime:         features.FlightTime,
        DwellTime:          features.DwellTime,
        ErrorRate:          features.ErrorRate,
        Rhythm:             features.Rhythm,
        Pressure:           features.Pressure,
        CollectedAt:        time.Now(),
    }

    return s.db.Create(featureData).Error
}

// 持续认证
func (s *BehaviorBiometricsService) ContinuousAuth(userID string, current *BehaviorData) (*AuthResult, error) {
    // 1. 获取历史基准
    baseline, err := s.getBaseline(userID)
    if err != nil {
        return &AuthResult{Confidence: 0, Anomaly: true}, err
    }

    // 2. 计算轨迹相似度
    trajectoryScore := s.trajectoryAnalyzer.Compare(
        baseline.Trajectory,
        current.Trajectory,
    )

    // 3. 计算键盘相似度
    keyboardScore := s.keyboardAnalyzer.Compare(
        baseline.Keyboard,
        current.Keyboard,
    )

    // 4. 集成评分
    finalScore := s.ensemble.Score([]float64{
        trajectoryScore,
        keyboardScore,
    })

    return &AuthResult{
        Confidence:  finalScore,
        Anomaly:     finalScore < 0.7,
        Recommendations: s.getRecommendations(finalScore),
    }, nil
}
```

### 案例 5.2: 日志规范与收集

**场景描述**

需要建立统一的日志规范，方便问题排查和安全审计。

**解决方案**

```go
type LogFormatter struct {
    serviceName string
    version     string
}

func NewLogFormatter(serviceName, version string) *LogFormatter {
    return &LogFormatter{
        serviceName: serviceName,
        version:     version,
    }
}

func (f *LogFormatter) Format(entry *logrus.Entry) ([]byte, error) {
    log := map[string]interface{}{
        "timestamp":   entry.Time.UTC().Format(time.RFC3339Nano),
        "level":       strings.ToUpper(entry.Level.String()),
        "service":     f.serviceName,
        "version":     f.version,
        "trace_id":    entry.Data["trace_id"],
        "span_id":     entry.Data["span_id"],
        "message":     entry.Message,
    }

    // 敏感字段脱敏
    sensitiveFields := []string{"password", "token", "secret", "key", "authorization"}
    for key, value := range entry.Data {
        if f.isSensitive(key) {
            log[key] = "***REDACTED***"
        } else {
            log[key] = value
        }
    }

    return json.Marshal(log)
}

func (f *LogFormatter) isSensitive(field string) bool {
    sensitive := []string{"password", "token", "secret", "key", "authorization", "credential"}
    fieldLower := strings.ToLower(field)

    for _, s := range sensitive {
        if strings.Contains(fieldLower, s) {
            return true
        }
    }

    return false
}

// 结构化日志示例
func (l *Logger) LogCaptchaVerify(req *VerifyRequest, result *VerifyResult) {

        "verify_result": result.Success,
        "verify_time_ms": result.VerifyTime.Milliseconds(),
        "ai_detected":   result.AIDetected,

}
```
