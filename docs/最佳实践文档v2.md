```go
// 缓存验证码结果
func CacheCaptchaResult(ctx context.Context, key string, result *VerifyResult) error {
    data, err := json.Marshal(result)
    if err != nil {
        return err
    }
    return redis.Set(ctx, key, data, 5*time.Minute).Err()
}

// 获取缓存结果
func GetCachedResult(ctx context.Context, key string) (*VerifyResult, error) {
    data, err := redis.Get(ctx, key).Bytes()
    if err != nil {
        return nil, err
    }
    
    var result VerifyResult
    if err := json.Unmarshal(data, &result); err != nil {
        return nil, err
    }
    return &result, nil
}
```

#### 多级缓存

```go
type MultiLevelCache struct {
    local  *lru.Cache    // L1: 本地内存
    redis  *redis.Client // L2: Redis
}

func (m *MultiLevelCache) Get(ctx context.Context, key string) ([]byte, error) {
    // L1查找
    if val, ok := m.local.Get(key); ok {
        return val.([]byte), nil
    }
    
    // L2查找
    val, err := m.redis.Get(ctx, key).Bytes()
    if err != nil {
        return nil, err
    }
    
    // 回填L1
    m.local.Add(key, val)
    
    return val, nil
}
```

## 安全加固

### 1. 签名验证

```go
func VerifySignature(secretKey string, params map[string]string, signature string) bool {
    // 1. 过滤空值和签名本身
    validParams := make(map[string]string)
    for k, v := range params {
        if v != "" && k != "signature" {
            validParams[k] = v
        }
    }
    
    // 2. 按字母顺序排序
    keys := make([]string, 0, len(validParams))
    for k := range validParams {
        keys = append(keys, k)
    }
    sort.Strings(keys)
    
    // 3. 拼接字符串
    var signString strings.Builder
    for _, k := range keys {
        signString.WriteString(k)
        signString.WriteString("=")
        signString.WriteString(validParams[k])
        signString.WriteString("&")
    }
    signString.WriteString("key=" + secretKey)
    
    // 4. 计算签名
    h := hmac.New(sha256.New, []byte(secretKey))
    h.Write([]byte(signString.String()))
    expectedSig := hex.EncodeToString(h.Sum(nil))
    
    return hmac.Equal([]byte(signature), []byte(expectedSig))
}
```

### 2. 请求限流

```go
func RateLimitMiddleware(limit int, window time.Duration) gin.HandlerFunc {
    requests := make(map[string][]time.Time)
    mutex := sync.Mutex{}
    
    return func(c *gin.Context) {
        clientIP := c.ClientIP()
        
        mutex.Lock()
        defer mutex.Unlock()
        
        now := time.Now()
        times := requests[clientIP]
        
        // 过滤过期请求
        cutoff := now.Add(-window)
        times = filter(times, func(t time.Time) bool {
            return t.After(cutoff)
        })
        
        if len(times) >= limit {
            c.AbortWithStatusJSON(http.StatusTooManyRequests, gin.H{
                "error": "Rate limit exceeded",
            })
            return
        }
        
        times = append(times, now)
        requests[clientIP] = times
        
        c.Next()
    }
}
```

```go
    // 验证轨迹点数量

        }
    }
    
    return nil
}
```

## 可靠性设计

### 1. 重试机制

```go
func RetryWithBackoff(fn func() error, maxRetries int) error {
    var err error
    for attempt := 0; attempt < maxRetries; attempt++ {
        err = fn()
        if err == nil {
            return nil
        }
        
        // 检查是否可重试
        if !isRetryable(err) {
            return err
        }
        
        // 指数退避
        backoff := time.Duration(math.Pow(2, float64(attempt))) * 100 * time.Millisecond
        maxBackoff := 30 * time.Second
        if backoff > maxBackoff {
            backoff = maxBackoff
        }
        
        time.Sleep(backoff)
    }
    return fmt.Errorf("max retries exceeded: %w", err)
}

func isRetryable(err error) bool {
    // 网络错误、超时、5xx错误可重试
    if strings.Contains(err.Error(), "timeout") {
        return true
    }
    if strings.Contains(err.Error(), "connection") {
        return true
    }
    return false
}
```

### 2. 熔断器

```go
type CircuitBreaker struct {
    failureThreshold int
    resetTimeout     time.Duration
    state            State
    failureCount     int
    lastFailure      time.Time
    mutex            sync.Mutex
}

type State int

const (
    StateClosed State = iota
    StateOpen
    StateHalfOpen
)

func (cb *CircuitBreaker) Execute(fn func() error) error {
    cb.mutex.Lock()
    defer cb.mutex.Unlock()
    
    switch cb.state {
    case StateOpen:
        if time.Since(cb.lastFailure) > cb.resetTimeout {
            cb.state = StateHalfOpen
        } else {
            return errors.New("circuit breaker is open")
        }
    case StateHalfOpen:
        // 允许请求通过测试
    }
    
    err := fn()
    if err != nil {
        cb.failureCount++
        cb.lastFailure = time.Now()
        
        if cb.failureCount >= cb.failureThreshold {
            cb.state = StateOpen
        }
        return err
    }
    
    cb.failureCount = 0
    cb.state = StateClosed
    return nil
}
```

### 3. 超时控制

```go
func TimeoutMiddleware(timeout time.Duration) gin.HandlerFunc {
    return func(c *gin.Context) {
        ctx, cancel := context.WithTimeout(c.Request.Context(), timeout)
        defer cancel()
        
        c.Request = c.Request.WithContext(ctx)
        
        done := make(chan struct{})
        
        go func() {
            c.Next()
            close(done)
        }()
        
        select {
        case <-done:
            // 正常完成
        case <-ctx.Done():
            c.AbortWithStatusJSON(http.StatusGatewayTimeout, gin.H{
                "error": "Request timeout",
            })
        }
    }
}
```

```go
type ShardingStrategy struct {
    shards   int
    hashFunc func(string) int
}

func NewShardingStrategy(shards int) *ShardingStrategy {
    return &ShardingStrategy{
        shards: shards,
        hashFunc: func(key string) int {
            h := fnv.New32a()
            h.Write([]byte(key))
            return int(h.Sum32()) % shards
        },
    }
}

func (s *ShardingStrategy) GetShard(key string) int {
    return s.hashFunc(key)
}

func (s *ShardingStrategy) GetDB(key string) int {
    return s.GetShard(key) / 4
}

func (s *ShardingStrategy) GetTable(key string) int {
```

### 3. 健康检查

```go
type HealthChecker struct {
    db    *sql.DB
    redis *redis.Client
}

func (hc *HealthChecker) Check(ctx context.Context) error {
    var errs []error
    
    // 数据库检查
    if err := hc.db.PingContext(ctx); err != nil {
        errs = append(errs, fmt.Errorf("database: %w", err))
    }
    
    // Redis检查
    if _, err := hc.redis.Ping(ctx).Result(); err != nil {
        errs = append(errs, fmt.Errorf("redis: %w", err))
    }
    
    if len(errs) > 0 {
        return fmt.Errorf("health check failed: %v", errs)
    }
    return nil
}

func (hc *HealthChecker) RegisterRoutes(r *gin.Engine) {
    r.GET("/health", func(c *gin.Context) {
        ctx, cancel := context.WithTimeout(c.Request.Context(), 5*time.Second)
        defer cancel()
        
        if err := hc.Check(ctx); err != nil {
            c.JSON(http.StatusServiceUnavailable, gin.H{
                "status": "unhealthy",
                "error": err.Error(),
            })
            return
        }
        
        c.JSON(http.StatusOK, gin.H{
            "status": "healthy",
        })
    })
}
```
