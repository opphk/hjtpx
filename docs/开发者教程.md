## 入门指南

### 什么是HJTPX

HJTPX是一个行为验证系统，提供多种验证码类型和高级行为分析能力，帮助保护您的应用免受自动化攻击。

### 核心功能

- **多类型验证码**: 滑块、点击、手势、3D、语音、Emoji、连连看
- **行为分析**: 高级人机识别
- **环境检测**: 设备指纹、Canvas/WebGL检测
- **风险评估**: 实时风险评分
- **无感验证**: 信任设备免验证

### 前置条件

- 基本的HTTP API调用经验
- 了解JSON数据格式
- 熟悉您所使用的编程语言

## API基础调用

### 完整验证流程

```bash
# 1. 获取验证码
CAPTCHA_RESP=$(curl -s http://localhost:8080/v3/captcha/slider)
SESSION_ID=$(echo $CAPTCHA_RESP | jq -r '.data.session_id')

echo "Session ID: $SESSION_ID"

# 2. 模拟用户滑动（实际应用中由前端完成）
# 用户滑动到位置 x=180, y=100

# 3. 提交验证
curl -X POST http://localhost:8080/v3/captcha/verify \
  -H "Content-Type: application/json" \
  -d '{
    "session_id": "'"$SESSION_ID"'",
    "type": "slider",
    "x": 180,
    "y": 100,
    "trajectory": [
      {"x": 0, "y": 100, "t": 0},
      {"x": 90, "y": 98, "t": 150},
      {"x": 180, "y": 101, "t": 300}
    ]
  }'
```

### 响应说明

| 字段 | 类型 | 说明 |
|------|------|------|
| `code` | integer | 状态码，0表示成功 |
| `message` | string | 状态消息 |
| `data.success` | boolean | 验证是否通过 |
| `data.message` | string | 验证结果说明 |
| `data.token` | string | 验证成功后的令牌 |
| `data.risk_score` | number | 风险分数(0-100) |

## SDK使用教程

### JavaScript SDK

#### 安装

```bash
npm install hjtpx-sdk
```

#### 基本用法

```javascript
import { CaptchaClient } from 'hjtpx-sdk';

// 创建客户端
const client = new CaptchaClient({
    baseUrl: 'http://localhost:8080',
    apiKey: 'your-api-key',
    timeout: 30000
});

// 获取滑块验证码
const slider = await client.getSliderCaptcha({
    width: 320,
    height: 160
});

console.log('Session ID:', slider.session_id);

// 验证验证码
const result = await client.verifyCaptcha({
    session_id: slider.session_id,
    type: 'slider',
    x: 180,
    y: slider.secret_y,
    trajectory: generateTrajectory()
});

if (result.success) {
    console.log('验证通过！Token:', result.token);
}
```

```bash
go get github.com/hjtpx/hjtpx/sdk/go
```

#### 基本用法

```go
package main

import (
    "github.com/hjtpx/hjtpx/sdk/go"
    "log"
)

func main() {
    // 创建客户端
    client := captcha.NewClient(
        captcha.WithAPIKey("your-api-key"),
        captcha.WithEndpoint("http://localhost:8080"),
    )
    defer client.Close()

    // 获取验证码
    slider, err := client.GenerateSliderCaptcha()
    if err != nil {
        log.Fatal(err)
    }

    // 验证
    result, err := client.VerifySliderCaptcha(slider.ChallengeID, "180")
    if err != nil {
        log.Fatal(err)
    }

    if result.Success {
        log.Printf("验证通过，风险分数: %.2f", result.Score)
    }
}
```

### Vue集成

```vue
<template>
  <form @submit.prevent="handleLogin">

    <CaptchaWidget
      ref="captcha"
      api-key="your-api-key"
      @verify="handleCaptchaVerify"
    />
    <button type="submit">登录</button>
  </form>
</template>

<script>
import { CaptchaWidget } from 'hjtpx-vue';

export default {
  components: { CaptchaWidget },
  data() {
    return {
      username: '',
      password: '',
      captchaToken: ''
    };
  },
  methods: {
    handleCaptchaVerify(token) {
      this.captchaToken = token;
    },
    async handleLogin() {
      // 发送登录请求
      const response = await fetch('/api/login', {
        method: 'POST',
        body: JSON.stringify({
          username: this.username,
          password: this.password,
          captcha_token: this.captchaToken
        })
      });
      console.log('登录结果:', await response.json());
    }
  }
};
</script>
```

## 后端集成

### 验证流程

```
用户前端 → 获取验证码 → 用户操作 → 提交验证 → 获取Token → 业务接口
                                                    ↓
                                            服务端验证Token
```

### Node.js后端验证

```javascript
const express = require('express');
const { CaptchaClient } = require('hjtpx-sdk');

const app = express();
app.use(express.json());

const captchaClient = new CaptchaClient({
    baseUrl: 'http://localhost:8080',
    apiKey: 'your-api-key'
});

// 登录接口
app.post('/api/login', async (req, res) => {
    const { username, password, captcha_token } = req.body;

    // 验证验证码Token
    try {
        const result = await captchaClient.verifyToken(captcha_token);
        
        if (!result.success) {
            return res.status(400).json({
                code: 400,
                message: '验证码验证失败'
            });
        }

        // 检查风险分数
        if (result.risk_score > 50) {
            console.warn(`高风险登录，风险分数: ${result.risk_score}`);
            // 可以要求额外验证
        }

        // 执行登录逻辑
        res.json({
            code: 0,
            message: '登录成功'
        });

    } catch (error) {
        res.status(500).json({
            code: 500,
            message: '验证码服务异常'
        });
    }
});

app.listen(3000, () => {
    console.log('服务器运行在 http://localhost:3000');
});
```

### Python后端验证

```python
from flask import Flask, request, jsonify
from hjtpx import CaptchaClient

app = Flask(__name__)
client = CaptchaClient(endpoint="http://localhost:8080", api_key="your-api-key")

@app.route('/api/login', methods=['POST'])
def login():
    data = request.get_json()
    
    # 验证验证码
    try:
        result = client.verify_token(data.get('captcha_token'))
        
        if not result.get('success'):
            return jsonify({'code': 400, 'message': '验证码验证失败'}), 400
        
        # 执行登录逻辑
        return jsonify({'code': 0, 'message': '登录成功'})
    
    except Exception as e:
        return jsonify({'code': 500, 'message': str(e)}), 500

if __name__ == '__main__':
    app.run(port=3000)
```

### Go后端验证

```go
package main

import (
    "encoding/json"
    "net/http"
    "github.com/hjtpx/hjtpx/sdk/go"
)

type LoginRequest struct {
    Username     string `json:"username"`
    Password     string `json:"password"`
    CaptchaToken string `json:"captcha_token"`
}

func loginHandler(w http.ResponseWriter, r *http.Request) {
    var req LoginRequest
    json.NewDecoder(r.Body).Decode(&req)

    // 验证验证码Token
    client := captcha.NewClient(
        captcha.WithAPIKey("your-api-key"),
        captcha.WithEndpoint("http://localhost:8080"),
    )
    defer client.Close()

    valid, err := client.VerifyToken(req.CaptchaToken)
    if err != nil || !valid {
        http.Error(w, "验证码验证失败", http.StatusBadRequest)
        return
    }

    // 执行登录逻辑
    w.Header().Set("Content-Type", "application/json")
    json.NewEncoder(w).Encode(map[string]interface{}{
        "code": 0,
        "message": "登录成功",
    })
}

func main() {
    http.HandleFunc("/api/login", loginHandler)
    http.ListenAndServe(":3000", nil)
}
```

## 高级功能

### 行为轨迹收集

```javascript
// 收集用户操作轨迹
function collectTrajectory() {
    const points = [];
    let startTime = Date.now();
    
    // 监听鼠标/触摸事件
    document.addEventListener('mousemove', (e) => {
        points.push({
            x: e.clientX,
            y: e.clientY,
            t: Date.now() - startTime
        });
    });
    
    return points;
}
```

## 常见问题

### Q: 验证码过期怎么办？

**A:** 验证码默认5分钟过期，过期后需要重新获取。

```javascript
// 自动刷新过期的验证码
captchaWidget.on('expired', () => {
    console.log('验证码已过期，正在刷新...');
    captchaWidget.refresh();
});
```

### Q: 如何处理高并发？

**A:** 使用连接池和限流：

```go
// 配置连接池
cfg := &captcha.Config{
    MaxOpenConns: 100,
    MaxIdleConns: 20,
}

// 使用限流器
limiter := rate.NewLimiter(100, 100) // 100 QPS
```

### Q: 验证总是失败？

**A:** 检查以下几点：
1. 轨迹数据是否真实自然
2. 滑动位置偏差是否在容差范围内
3. Session ID是否正确
4. 请求是否在有效期内

### Q: 如何自定义验证码样式？

**A:** 使用白标配置：

```javascript
await client.updateWhitelabelConfig({
    brand_name: 'YourBrand',
    primary_color: '#722ed1',
    custom_css: `.captcha-container { border-radius: 12px; }`
});
```
