## 快速开始

### 1. 安装SDK

根据您的编程语言选择合适的SDK：

```bash
# Go
go get github.com/hjtpx/hjtpx/sdk/go

# Python
pip install hjtpx-captcha

# JavaScript/TypeScript
npm install hjtpx-sdk

# Java
<dependency>
    <groupId>com.hjtpx</groupId>
    <artifactId>captcha-sdk</artifactId>
    <version>1.0.0</version>
</dependency>

# C#
dotnet add package Hjtpx.Captcha.Sdk
```

### 2. 基本使用

```go
package main

import (
    "fmt"
    captcha "github.com/hjtpx/hjtpx/sdk/go"
)

func main() {
    // 创建客户端
    client := captcha.NewCaptchaClient("app-id", "app-secret", nil)
    defer client.Close()

    // 生成滑块验证码
    slider, err := client.GenerateSliderCaptcha()
    if err != nil {
        panic(err)
    }
    fmt.Printf("Session ID: %s\n", slider.ChallengeID)

    // 验证验证码
    result, err := client.VerifySliderCaptcha(slider.ChallengeID, "150")
    if err != nil {
        panic(err)
    }
    fmt.Printf("Verification: %v\n", result.Success)
}
```

### 后端集成

#### Go后端

```go
package main

import (
    "net/http"
    "github.com/gin-gonic/gin"
    captcha "github.com/hjtpx/hjtpx/sdk/go"
)

func main() {
    r := gin.Default()

    // 初始化验证码客户端
    client := captcha.NewCaptchaClient("app-id", "app-secret", nil)

    // 获取滑块验证码
    r.GET("/captcha/slider", func(c *gin.Context) {
        slider, err := client.GenerateSliderCaptcha()
        if err != nil {
            c.JSON(http.StatusInternalServerError, gin.H{
                "success": false,
                "error": err.Error(),
            })
            return
        }

        c.JSON(http.StatusOK, gin.H{
            "success": true,
            "data": slider,
        })
    })

    // 验证验证码
    r.POST("/captcha/verify", func(c *gin.Context) {
        var req struct {
            SessionID string `json:"session_id"`
            Type      string `json:"type"`
            X         int    `json:"x"`
        }

        if err := c.ShouldBindJSON(&req); err != nil {
            c.JSON(http.StatusBadRequest, gin.H{
                "success": false,
                "error": "Invalid request",
            })
            return
        }

        result, err := client.VerifySliderCaptcha(req.SessionID, fmt.Sprintf("%d", req.X))
        if err != nil {
            c.JSON(http.StatusInternalServerError, gin.H{
                "success": false,
                "error": err.Error(),
            })
            return
        }

        c.JSON(http.StatusOK, gin.H{
            "success": result.Success,
            "score": result.Score,
        })
    })

    r.Run(":8080")
}
```

## 框架集成

### React集成

```jsx
import React, { useState, useEffect } from 'react';
import axios from 'axios';

function CaptchaSlider({ onVerify }) {
    const [captcha, setCaptcha] = useState(null);
    const [position, setPosition] = useState(0);
    const [trajectory, setTrajectory] = useState([]);

    useEffect(() => {
        loadCaptcha();
    }, []);

    const loadCaptcha = async () => {
        const response = await axios.get('/api/captcha/slider');
        setCaptcha(response.data.data);
    };

    const handleDrag = (e) => {
        const rect = e.target.getBoundingClientRect();
        const x = e.clientX - rect.left;
        setPosition(x);
        setTrajectory([...trajectory, { x, t: Date.now() }]);
    };

    const handleVerify = async () => {
        const response = await axios.post('/api/captcha/verify', {
            session_id: captcha.session_id,
            type: 'slider',
            x: position,
            trajectory: trajectory
        });
        onVerify(response.data.success);
    };

    if (!captcha) return <div>Loading...</div>;

    return (
        <div className="captcha-container">
            <img src={captcha.image_url} alt="Background" />
            <div 
                className="slider" 
                style={{ left: position }}
                onMouseMove={handleDrag}
                onMouseUp={handleVerify}
            >
                <img src={captcha.puzzle_url} alt="Slider" />
            </div>
        </div>
    );
}
```

### Vue集成

```vue
<template>
  <div class="captcha-container">
    <img :src="captcha.image_url" />
    <div 
      class="slider" 
      :style="{ left: sliderPosition + 'px' }"
      @mousedown="startDrag"
    >
      <img :src="captcha.puzzle_url" />
    </div>
    <button @click="verify">验证</button>
  </div>
</template>

<script>
export default {
  data() {
    return {
      captcha: null,
      sliderPosition: 0,
      trajectory: [],
      isDragging: false
    }
  },
  async mounted() {
    const response = await this.$axios.get('/api/captcha/slider');
    this.captcha = response.data.data;
  },
  methods: {
    startDrag(e) {
      this.isDragging = true;
      this.startX = e.clientX;
      document.addEventListener('mousemove', this.onDrag);
      document.addEventListener('mouseup', this.endDrag);
    },
    onDrag(e) {
      if (!this.isDragging) return;
      this.sliderPosition = e.clientX - this.startX;
      this.trajectory.push({
        x: this.sliderPosition,
        t: Date.now()
      });
    },
    endDrag() {
      this.isDragging = false;
      document.removeEventListener('mousemove', this.onDrag);
      document.removeEventListener('mouseup', this.endDrag);
    },
    async verify() {
      const response = await this.$axios.post('/api/captcha/verify', {
        session_id: this.captcha.session_id,
        type: 'slider',
        x: this.sliderPosition,
        trajectory: this.trajectory
      });
      if (response.data.success) {
        this.$emit('verify');
      }
    }
  }
}
</script>
```

### 2. 错误处理

```javascript
async function verifyCaptcha(sessionId, x, trajectory) {
    try {
        const response = await fetch('/api/captcha/verify', {
            method: 'POST',
            headers: {'Content-Type': 'application/json'},
            body: JSON.stringify({ session_id: sessionId, x, trajectory })
        });

        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }

        const data = await response.json();

        if (data.code !== 0) {
            throw new Error(data.message || 'Verification failed');
        }

        return data.data;
    } catch (error) {
        console.error('Verification error:', error);
        
        // 区分错误类型
        if (error.message.includes('timeout')) {
            // 处理超时
            return handleTimeout();
        } else if (error.message.includes('network')) {
            // 处理网络错误
            return handleNetworkError();
        } else {
            // 处理其他错误
            return handleError(error);
        }
    }
}
```
