```javascript
// 数据缓存实现
const DATA_CACHE_DURATION = 10000;
const PREFETCH_INTERVAL = 30000;
let dataCache = new Map();

// 缓存获取
dataCache.get = function(key) {
    const item = Map.prototype.get.call(this, key);
    if (!item) return null;
    
    if (Date.now() - item.timestamp > DATA_CACHE_DURATION) {
        this.delete(key);
        return null;
    }
    return item.data;
};
```
