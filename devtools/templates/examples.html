class HJTPXCaptcha {

        this.appId = appId;
        this.appKey = appKey;
    }
    
    async getSliderCaptcha() {
        const response = await fetch(\`\${this.apiUrl}/api/v1/captcha/slider\`);
        if (!response.ok) throw new Error('Request failed');
        return await response.json();
    }
    
    async verifySliderCaptcha(sessionId, x) {
        const response = await fetch(\`\${this.apiUrl}/api/v1/captcha/verify\`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                session_id: sessionId,
                answer: { x: x }
            })
        });
        if (!response.ok) throw new Error('Request failed');
        return await response.json();
    }
    
    async getClickCaptcha() {
        const response = await fetch(\`\${this.apiUrl}/api/v1/captcha/click\`);

        const response = await fetch(\`\${this.apiUrl}/api/v1/captcha/verify\`, {

            body: JSON.stringify({
                session_id: sessionId,
                answer: { points: points }
            })
        });
        if (!response.ok) throw new Error('Request failed');
        return await response.json();
    }
}

Go 完整集成示例 
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

type HJTPXCaptcha struct {
    APIUrl string
    AppID  string
    AppKey string
}

func NewHJTPXCaptcha(apiUrl, appID, appKey string) *HJTPXCaptcha {
    return &HJTPXCaptcha{
        APIUrl: apiUrl,
        AppID:  appID,
        AppKey: appKey,
    }
}

func (c *HJTPXCaptcha) GetSliderCaptcha() (map[string]interface{}, error) {
    resp, err := http.Get(c.APIUrl + "/api/v1/captcha/slider")
    if err != nil {
        return nil, err
    }
    defer resp.Body.Close()
    
    body, err := ioutil.ReadAll(resp.Body)
    if err != nil {
        return nil, err
    }
    
    var result map[string]interface{}
    err = json.Unmarshal(body, &result)
    return result, err
}

func (c *HJTPXCaptcha) VerifySliderCaptcha(sessionID string, x int) (map[string]interface{}, error) {
    data := map[string]interface{}{
        "session_id": sessionID,
        "answer": map[string]interface{}{
            "x": x,
        },
    }
    
    jsonData, err := json.Marshal(data)
    if err != nil {
        return nil, err
    }
    
    resp, err := http.Post(c.APIUrl+"/api/v1/captcha/verify", "application/json", bytes.NewBuffer(jsonData))
    if err != nil {
        return nil, err
    }
    defer resp.Body.Close()
    
    body, err := ioutil.ReadAll(resp.Body)
    if err != nil {
        return nil, err
    }
    
    var result map[string]interface{}
    err = json.Unmarshal(body, &result)
    return result, err
}

func main() {
    captcha := NewHJTPXCaptcha("http://localhost:8080", "your_app_id", "your_app_key")
    
    sliderData, err := captcha.GetSliderCaptcha()
    if err != nil {
        fmt.Printf("错误: %v\n", err)
        return
    }
    fmt.Printf("滑动验证码: %v\n", sliderData)
    
    result, err := captcha.VerifySliderCaptcha(sliderData["session_id"].(string), 50)
    if err != nil {
        fmt.Printf("错误: %v\n", err)
        return
    }
    fmt.Printf("验证结果: %v\n", result)
}

import java.net.URI;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.util.HashMap;
import java.util.Map;
import com.fasterxml.jackson.databind.ObjectMapper;

public class HJTPXCaptcha {
    private String apiUrl;
    private String appId;
    private String appKey;
    private HttpClient client;
    private ObjectMapper mapper;
    
    public HJTPXCaptcha(String apiUrl, String appId, String appKey) {
        this.apiUrl = apiUrl;
        this.appId = appId;
        this.appKey = appKey;
        this.client = HttpClient.newHttpClient();
        this.mapper = new ObjectMapper();
    }
    
    public Map getSliderCaptcha() throws Exception {
        HttpRequest request = HttpRequest.newBuilder()
            .uri(URI.create(apiUrl + "/api/v1/captcha/slider"))
            .GET()
            .build();
        
        HttpResponse response = client.send(request, HttpResponse.BodyHandlers.ofString());
        return mapper.readValue(response.body(), Map.class);
    }
    
    public Map verifySliderCaptcha(String sessionId, int x) throws Exception {
        Map data = new HashMap<>();
        data.put("session_id", sessionId);
        Map answer = new HashMap<>();
        answer.put("x", x);
        data.put("answer", answer);
        
        String jsonBody = mapper.writeValueAsString(data);
        
        HttpRequest request = HttpRequest.newBuilder()
            .uri(URI.create(apiUrl + "/api/v1/captcha/verify"))
            .header("Content-Type", "application/json")
            .POST(HttpRequest.BodyPublishers.ofString(jsonBody))
            .build();
        
        HttpResponse response = client.send(request, HttpResponse.BodyHandlers.ofString());
        return mapper.readValue(response.body(), Map.class);
    }
    
    public static void main(String[] args) {
        try {
            HJTPXCaptcha captcha = new HJTPXCaptcha("http://localhost:8080", "your_app_id", "your_app_key");
            
            Map sliderData = captcha.getSliderCaptcha();
            System.out.println("滑动验证码: " + sliderData);
            
            Map result = captcha.verifySliderCaptcha((String)sliderData.get("session_id"), 50);
            System.out.println("验证结果: " + result);
        } catch (Exception e) {
            System.err.println("错误: " + e.getMessage());
            e.printStackTrace();
        }
    }
}
