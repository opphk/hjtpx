滑块验证演示

拖动滑块将拼图块移动到正确位置，即可完成验证。

API调用示例

// 生成滑块验证码
fetch('/api/v1/captcha/slider/generate', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json'
  }
})
.then(res => res.json())
.then(data => {
  console.log(data);
  // {
  //   session_id: "xxx",
  //   image_url: "/captcha/slider/xxx.png",
  //   puzzle_y: 50
  // }
});

// 验证滑块
fetch('/api/v1/captcha/slider/verify', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    session_id: 'xxx',
    x: 200,
    y: 50
  })
})
.then(res => res.json())
.then(data => {
  console.log(data);
  // { success: true }
});

快速集成示例

只需几行代码即可将验证码集成到您的网站。

HTML

<!-- 引入验证码脚本 -->
<script src="https://example.com/captcha.js"></script>

<!-- 创建验证码容器 -->
<div id="captcha-container"></div>

<!-- 初始化验证码 -->
<script>
const captcha = new Captcha('captcha-container', {
  apiBase: '/api/v1',
  type: 'slider',
  onSuccess: function(result) {
    // 验证成功处理
    console.log(result.session_id);
  },
  onError: function(error) {
    // 验证失败处理
    console.error(error);
  }
});
</script>

React

import { useEffect, useRef } from 'react';

function CaptchaComponent() {
  const containerRef = useRef(null);

  useEffect(() => {
    const captcha = new Captcha(containerRef.current, {
      apiBase: '/api/v1',
      type: 'slider',
      onSuccess: (result) => {
        // 处理验证成功
        console.log(result.session_id);
      },
      onError: (error) => {
        // 处理验证失败
        console.error(error);
      }
    });

    return () => {
      captcha.destroy();
    };
  }, []);

  return <div ref={containerRef}></div>;
}
